### 2. 用户与认证路由：`backend/src/routes/users.ts`

```typescript
import crypto from 'crypto';
import { Router } from 'express';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { db } from '../db/init';

export const usersRouter = Router();

const JWT_SECRET = process.env.JWT_SECRET || 'dev-secret';
const JWT_EXPIRE = process.env.JWT_EXPIRE || '7d';
const CODE_EXPIRE_MINUTES = 5;

function randomCode(): string {
  return String(Math.floor(100000 + Math.random() * 900000));
}

// 发送验证码：邮箱 或 手机号。当前未接短信/邮件服务时，响应中返回 code 供测试。
usersRouter.post('/send-code', (req, res) => {
  const { email, phone } = req.body || {};
  const usePhone = phone != null && String(phone).trim() !== '';
  const useEmail = email != null && String(email).trim() !== '';
  if ((!usePhone && !useEmail) || (usePhone && useEmail)) {
    return res.status(400).json({ error: '请只填邮箱或只填手机号其一' });
  }
  const target = usePhone ? String(phone).trim() : String(email).trim();
  const type = usePhone ? 'phone' : 'email';
  if (usePhone && !/^1\d{10}$/.test(target)) {
    return res.status(400).json({ error: '请输入正确的 11 位手机号' });
  }
  const code = randomCode();
  const expiresAt = new Date(Date.now() + CODE_EXPIRE_MINUTES * 60 * 1000).toISOString();
  db.prepare('DELETE FROM verification_codes WHERE target = ? AND type = ?').run(target, type);
  db.prepare('INSERT INTO verification_codes (target, type, code, expires_at) VALUES (?, ?, ?, ?)').run(target, type, code, expiresAt);
  res.json({ message: '验证码已发送', code: process.env.NODE_ENV !== 'production' ? code : undefined });
});

// 注册：邮箱 或 手机号 + 验证码 + 密码，返回 JWT
usersRouter.post('/register', (req, res) => {
  try {
    const { email, phone, code, password } = req.body || {};
    const usePhone = phone != null && String(phone).trim() !== '';
    const useEmail = email != null && String(email).trim() !== '';
    if ((!usePhone && !useEmail) || !code || !password) {
      return res.status(400).json({ error: '请提供邮箱或手机号、验证码和密码（至少 6 位）' });
    }
    if (usePhone && useEmail) {
      return res.status(400).json({ error: '请只填邮箱或只填手机号其一' });
    }
    if (String(password).length < 6) {
      return res.status(400).json({ error: '密码至少 6 位字符' });
    }
    const target = usePhone ? String(phone).trim() : String(email).trim();
    const type = usePhone ? 'phone' : 'email';
    const row = db.prepare(
      'SELECT code, expires_at FROM verification_codes WHERE target = ? AND type = ? ORDER BY id DESC LIMIT 1'
    ).get(target, type) as { code: string; expires_at: string } | undefined;
    if (!row || row.code !== String(code).trim()) {
      return res.status(400).json({ error: '验证码错误' });
    }
    if (new Date(row.expires_at) < new Date()) {
      return res.status(400).json({ error: '验证码已过期，请重新获取' });
    }
    const password_hash = bcrypt.hashSync(String(password), 10);
    db.prepare('DELETE FROM verification_codes WHERE target = ? AND type = ?').run(target, type);
    if (usePhone) {
      const p = target;
      const stmt = db.prepare(
        'INSERT INTO users (phone, email, password_hash) VALUES (?, NULL, ?)'
      );
      const result = stmt.run(p, password_hash);
      const id = Number(result.lastInsertRowid);
      const token = jwt.sign({ userId: id, phone: p }, JWT_SECRET, { expiresIn: JWT_EXPIRE });
      return res.status(201).json({ token, userId: id, phone: p });
    }
    const e = target;
    const stmt = db.prepare(
      'INSERT INTO users (email, password_hash) VALUES (?, ?)'
    );
    const result = stmt.run(e, password_hash);
    const id = Number(result.lastInsertRowid);
    const token = jwt.sign({ userId: id, email: e }, JWT_SECRET, { expiresIn: JWT_EXPIRE });
    return res.status(201).json({ token, userId: id, email: e });
  } catch (e: unknown) {
    const msg = e instanceof Error ? e.message : String(e);
    if (msg.includes('UNIQUE') || msg.includes('constraint')) {
      return res.status(409).json({ error: '该邮箱或手机号已注册' });
    }
    console.error('register error', e);
    return res.status(500).json({ error: '注册失败，请稍后再试' });
  }
});

// 登录：邮箱 或 手机号 + 密码，返回 JWT
usersRouter.post('/login', (req, res) => {
  const { email, phone, password } = req.body;
  const usePhone = phone != null && String(phone).trim() !== '';
  const useEmail = email != null && String(email).trim() !== '';
  if ((!usePhone && !useEmail) || !password) {
    return res.status(400).json({ error: '请提供邮箱或手机号，以及密码' });
  }
  if (usePhone && useEmail) {
    return res.status(400).json({ error: '请只填邮箱或只填手机号其一' });
  }
  const row = db.prepare(
    usePhone
      ? 'SELECT id, password_hash, phone FROM users WHERE phone = ?'
      : 'SELECT id, password_hash, email FROM users WHERE email = ?'
  ).get(usePhone ? String(phone).trim() : String(email).trim()) as { id: number; password_hash: string; email?: string; phone?: string } | undefined;
  if (!row) {
    return res.status(401).json({ error: '账号或密码错误' });
  }
  const ok = bcrypt.compareSync(String(password), row.password_hash);
  if (!ok) {
    return res.status(401).json({ error: '账号或密码错误' });
  }
  const payload = usePhone ? { userId: row.id, phone: row.phone } : { userId: row.id, email: row.email };
  const token = jwt.sign(payload, JWT_SECRET, { expiresIn: JWT_EXPIRE });
  if (usePhone) {
    return res.json({ token, userId: row.id, phone: row.phone });
  }
  return res.json({ token, userId: row.id, email: row.email });
});

const FRONTEND_URL = process.env.FRONTEND_URL || 'http://localhost:5173';
const RESET_TOKEN_EXPIRE_MINUTES = 60;

function randomToken(): string {
  return crypto.randomBytes(32).toString('hex');
}

// 忘记密码（仅支持邮箱注册用户）：生成重置链接，未配置发信时直接返回链接供测试
usersRouter.post('/forgot-password', (req, res) => {
  const { email } = req.body;
  const e = email != null ? String(email).trim() : '';
  if (!e) return res.status(400).json({ error: '请提供邮箱' });
  const row = db.prepare('SELECT id FROM users WHERE email = ?').get(e) as { id: number } | undefined;
  if (!row) {
    return res.status(404).json({ error: '该邮箱未注册' });
  }
  const token = randomToken();
  const expiresAt = new Date(Date.now() + RESET_TOKEN_EXPIRE_MINUTES * 60 * 1000).toISOString();
  db.prepare('DELETE FROM password_reset_tokens WHERE email = ?').run(e);
  db.prepare('INSERT INTO password_reset_tokens (email, token, expires_at) VALUES (?, ?, ?)').run(e, token, expiresAt);
  const resetLink = `${FRONTEND_URL}/reset-password?token=${encodeURIComponent(token)}`;
  res.json({
    message: '已生成重置链接',
    resetLink,
  });
});

// 重置密码：凭 token 设置新密码
usersRouter.post('/reset-password', (req, res) => {
  const { token, newPassword } = req.body;
  if (!token || typeof token !== 'string') return res.status(400).json({ error: '请提供重置 token' });
  if (!newPassword || String(newPassword).length < 6) return res.status(400).json({ error: '新密码至少 6 位' });
  const row = db.prepare(
    'SELECT email, expires_at FROM password_reset_tokens WHERE token = ?'
  ).get(token.trim()) as { email: string; expires_at: string } | undefined;
  if (!row) return res.status(400).json({ error: '链接无效或已失效' });
  if (new Date(row.expires_at) < new Date()) {
    db.prepare('DELETE FROM password_reset_tokens WHERE token = ?').run(token.trim());
    return res.status(400).json({ error: '链接已过期，请重新申请' });
  }
  const password_hash = bcrypt.hashSync(String(newPassword), 10);
  db.prepare('UPDATE users SET password_hash = ? WHERE email = ?').run(password_hash, row.email);
  db.prepare('DELETE FROM password_reset_tokens WHERE token = ?').run(token.trim());
  res.json({ message: '密码已重置，请使用新密码登录' });
});
```

### 3. JWT 鉴权中间件：`backend/src/middleware/auth.ts`

```typescript
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

const secret = process.env.JWT_SECRET || 'dev-secret';

export interface JwtPayload {
  userId: number;
  email?: string;
}

export interface AuthRequest extends Request {
  user?: JwtPayload;
}

/**
 * 校验 Authorization: Bearer <token>，将 userId 写入 req.user
 */
export function requireAuth(req: AuthRequest, res: Response, next: NextFunction): void {
  const auth = req.headers.authorization;
  if (!auth || !auth.startsWith('Bearer ')) {
    res.status(401).json({ error: '请先登录' });
    return;
  }
  const token = auth.slice(7);
  try {
    const payload = jwt.verify(token, secret) as JwtPayload;
    req.user = payload;
    next();
  } catch {
    res.status(401).json({ error: '登录已过期或无效，请重新登录' });
  }
}

/**
 * 若带 token 则解析并写入 req.user，不强制登录
 */
export function optionalAuth(req: AuthRequest, _res: Response, next: NextFunction): void {
  const auth = req.headers.authorization;
  if (auth?.startsWith('Bearer ')) {
    try {
      req.user = jwt.verify(auth.slice(7), secret) as JwtPayload;
    } catch {
      // 忽略无效 token
    }
  }
  next();
}
```

### 2. 认证上下文：`frontend/src/contexts/AuthContext.tsx`

```tsx
import { createContext, useContext, useState, useCallback } from 'react';

const TOKEN_KEY = 'zchoose_token';
const USER_KEY = 'zchoose_user';
export const REMEMBER_KEY = 'zchoose_remember';

export interface User {
  userId: number;
  email?: string;
  phone?: string;
}

export interface RememberedCredentials {
  account: string;
  password: string;
  type: 'email' | 'phone';
}

interface AuthContextValue {
  token: string | null;
  user: User | null;
  login: (account: string, password: string, type: 'email' | 'phone') => Promise<void>;
  register: (account: string, code: string, password: string, type: 'email' | 'phone') => Promise<void>;
  logout: () => void;
  isAuthenticated: boolean;
}

const AuthContext = createContext<AuthContextValue | null>(null);

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [token, setToken] = useState<string | null>(() => localStorage.getItem(TOKEN_KEY));
  const [user, setUser] = useState<User | null>(() => {
    const raw = localStorage.getItem(USER_KEY);
    return raw ? JSON.parse(raw) : null;
  });

  const login = useCallback(async (account: string, password: string, type: 'email' | 'phone') => {
    const body = type === 'phone' ? { phone: account.trim(), password } : { email: account.trim(), password };
    const res = await fetch('/api/users/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    if (!res.ok) {
      const data = await res.json().catch(() => ({}));
      throw new Error(data.error || '登录失败');
    }
    const data = await res.json();
    const userData = { userId: data.userId, ...(data.email != null && { email: data.email }), ...(data.phone != null && { phone: data.phone }) };
    setToken(data.token);
    setUser(userData);
    localStorage.setItem(TOKEN_KEY, data.token);
    localStorage.setItem(USER_KEY, JSON.stringify(userData));
  }, []);

  const register = useCallback(async (account: string, code: string, password: string, type: 'email' | 'phone') => {
    const body = type === 'phone'
      ? { phone: account.trim(), code: code.trim(), password }
      : { email: account.trim(), code: code.trim(), password };
    const res = await fetch('/api/users/register', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    if (!res.ok) {
      const data = await res.json().catch(() => ({}));
      const msg = (data && (data.error || data.message)) || (res.status === 409 ? '该账号已注册' : res.status === 400 ? '请检查邮箱/手机号和密码格式' : '注册失败');
      throw new Error(msg);
    }
    const data = await res.json();
    const userData = { userId: data.userId, ...(data.email != null && { email: data.email }), ...(data.phone != null && { phone: data.phone }) };
    setToken(data.token);
    setUser(userData);
    localStorage.setItem(TOKEN_KEY, data.token);
    localStorage.setItem(USER_KEY, JSON.stringify(userData));
  }, []);

  const logout = useCallback(() => {
    setToken(null);
    setUser(null);
    localStorage.removeItem(TOKEN_KEY);
    localStorage.removeItem(USER_KEY);
  }, []);

  return (
    <AuthContext.Provider
      value={{
        token,
        user,
        login,
        register,
        logout,
        isAuthenticated: !!token,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used within AuthProvider');
  return ctx;
}
```

### 5. 顶栏（唤起登录/显示用户/退出）：`frontend/src/components/TopBar.tsx`

```tsx
import { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { useAuth } from '../contexts/AuthContext';
import LoginModal from './LoginModal';
import './TopBar.css';

export default function TopBar() {
  const { user, logout, isAuthenticated } = useAuth();
  const [showLogin, setShowLogin] = useState(false);
  const [points, setPoints] = useState<number | null>(null);

  useEffect(() => {
    if (!user?.userId) {
      setPoints(null);
      return;
    }
    fetch(`/api/points/${user.userId}`)
      .then((res) => res.ok ? res.json() : null)
      .then((data) => data != null && typeof data.points === 'number' ? setPoints(data.points) : setPoints(null))
      .catch(() => setPoints(null));
  }, [user?.userId]);

  return (
    <>
      <header className="topbar">
        <Link to="/home" className="topbar__logo">Zchoose</Link>
        <div className="topbar__right">
          {isAuthenticated && (
            <span className="topbar__points">积分 {points != null ? points : '--'}</span>
          )}
          {isAuthenticated ? (
            <div className="topbar__user-wrap">
              <span className="topbar__user-email">{user?.email ?? user?.phone ?? ''}</span>
              <button type="button" className="topbar__user" onClick={logout}>退出</button>
            </div>
          ) : (
            <button type="button" className="topbar__user" onClick={() => setShowLogin(true)}>登录</button>
          )}
        </div>
      </header>
      {showLogin && <LoginModal onClose={() => setShowLogin(false)} />}
    </>
  );
}
```

### 6. 重置密码页：`frontend/src/pages/ResetPassword.tsx`

```tsx
import { useState } from 'react';
import { useSearchParams, useNavigate } from 'react-router-dom';
import './ResetPassword.css';

export default function ResetPassword() {
  const [searchParams] = useSearchParams();
  const navigate = useNavigate();
  const token = searchParams.get('token') ?? '';
  const [password, setPassword] = useState('');
  const [confirm, setConfirm] = useState('');
  const [error, setError] = useState('');
  const [loading, setLoading] = useState(false);
  const [done, setDone] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setError('');
    if (!token) {
      setError('缺少重置链接，请从邮件中的链接进入');
      return;
    }
    if (password.length < 6) {
      setError('密码至少 6 位');
      return;
    }
    if (password !== confirm) {
      setError('两次输入的密码不一致');
      return;
    }
    setLoading(true);
    try {
      const res = await fetch('/api/users/reset-password', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ token, newPassword: password }),
      });
      const data = await res.json().catch(() => ({}));
      if (!res.ok) throw new Error(data.error || '重置失败');
      setDone(true);
    } catch (err) {
      setError(err instanceof Error ? err.message : '重置失败');
    } finally {
      setLoading(false);
    }
  };

  if (done) {
    return (
      <div className="reset-password">
        <p className="reset-password__success">密码已重置，请使用新密码登录。</p>
        <button type="button" className="reset-password__btn" onClick={() => navigate('/home')}>返回首页</button>
      </div>
    );
  }

  if (!token) {
    return (
      <div className="reset-password">
        <p className="reset-password__error">无效的重置链接，请在登录页重新申请「忘记密码」。</p>
        <button type="button" className="reset-password__btn" onClick={() => navigate('/home')}>返回首页</button>
      </div>
    );
  }

  return (
    <div className="reset-password">
      <h2 className="reset-password__title">设置新密码</h2>
      <form onSubmit={handleSubmit} className="reset-password__form">
        <label className="reset-password__label">
          新密码
          <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} placeholder="至少 6 位" className="reset-password__input" autoComplete="new-password" />
        </label>
        <label className="reset-password__label">
          确认密码
          <input type="password" value={confirm} onChange={(e) => setConfirm(e.target.value)} placeholder="再次输入新密码" className="reset-password__input" autoComplete="new-password" />
        </label>
        {error && <p className="reset-password__error">{error}</p>}
        <button type="submit" className="reset-password__btn reset-password__btn--primary" disabled={loading}>
          {loading ? '提交中...' : '确认重置'}
        </button>
      </form>
    </div>
  );
}
```

| 方法 | 路径 | 说明 |
|-----|------|------|
| POST | `/api/users/send-code` | 发送验证码（body: `email` 或 `phone`） |
| POST | `/api/users/register` | 注册（body: `email`/`phone` + `code` + `password`） |
| POST | `/api/users/login` | 登录（body: `email`/`phone` + `password`） |
| POST | `/api/users/forgot-password` | 忘记密码（body: `email`），返回 `resetLink` |
| POST | `/api/users/reset-password` | 重置密码（body: `token` + `newPassword`） |

需要登录的接口在请求头中携带：`Authorization: Bearer <token>`，由 `requireAuth` 或 `optionalAuth` 中间件校验。
